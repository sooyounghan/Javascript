function listAncestors(obj) {
  let curObj = obj;
  let constNames = [];
  while (Object.getPrototypeOf(curObj)) {
    curObj = Object.getPrototypeOf(curObj);
    constNames.push(curObj.constructor.name);
  }
  return constNames.join(', ');
}
